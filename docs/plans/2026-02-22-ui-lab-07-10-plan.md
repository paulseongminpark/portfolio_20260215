function WorkRow({
  item,
  index,
  isLast,
  onSelect,
}: {
  item: { id: string; title: string; period: string; tag: string; description: string };
  index: number;
  isLast?: boolean;
  onSelect?: () => void;
}) {
  const [hovered, setHovered] = useState(false);
  const num = String(index + 1).padStart(2, "0");

  return (
    <div
      onMouseEnter={() => setHovered(true)}
      onMouseLeave={() => setHovered(false)}
      onClick={onSelect}
      style={{
        display: "grid",
        gridTemplateColumns: "28px 1fr auto",
        gap: "0 24px",
        alignItems: "start",
        padding: "28px 0",
        borderBottom: isLast ? "none" : `1px solid ${colors.border}`,
        cursor: onSelect ? "pointer" : "default",
        opacity: hovered && onSelect ? 0.7 : 1,
        transition: "opacity 0.2s",
      }}
    >
      {/* 번호 */}
      <span
        style={{
          fontFamily: "'Instrument Sans', sans-serif",
          fontSize: 11,
          fontWeight: 600,
          color: colors.accent,
          paddingTop: 4,
          opacity: 0.7,
        }}
      >
        {num}
      </span>

      <div>
        {/* 태그 */}
        <div
          style={{
            fontFamily: "'Instrument Sans', sans-serif",
            fontSize: 11,
            color: colors.sub,
            letterSpacing: 1,
            marginBottom: 10,
          }}
        >
          {item.tag}
        </div>
        {/* 제목 */}
        <div
          style={{
            fontFamily: "'Libre Baskerville', serif",
            fontSize: 22,
            fontWeight: 400,
            color: colors.text,
            marginBottom: 12,
            lineHeight: 1.25,
            fontStyle: hovered && onSelect ? "italic" : "normal",
            transition: "font-style 0.1s",
          }}
        >
          {item.title}
        </div>
        {/* 설명 */}
        <p
          style={{
            fontFamily: "'Instrument Sans', sans-serif",
            fontSize: 13,
            lineHeight: 1.7,
            color: colors.sub,
            margin: 0,
            maxWidth: 480,
          }}
        >
          {item.description}
        </p>
      </div>

      {/* 기간 */}
      <div
        style={{
          fontFamily: "'Instrument Sans', sans-serif",
          fontSize: 12,
          color: colors.sub,
          whiteSpace: "nowrap",
          paddingTop: 4,
        }}
      >
        {item.period}
      </div>
    </div>
  );
}
